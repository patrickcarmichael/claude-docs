## Step 6: Query the index

query = "What did Microsoft announce in Las Vegas?"

xq = co.embed(
    texts=[query],
    model='embed-english-v3.0',
    input_type='search_query',
    truncate='END'
).embeddings

print(np.array(xq).shape)

res = idx.query(xq, top_k=20, include_metadata=True)
